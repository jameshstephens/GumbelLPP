\section{Auxiliary Functions}

\begin{definition}
  \label{def:h_N}
  \lean{h}
  \leanok
  For a real number $N$, we define the function $h_N : \mathbb{R} \to \mathbb{R}$ by
  \[
    h_N(x) = -\log\left(N \left(1 - \exp\left(-\frac{e^{-x}}{N}\right)\right)\right) - x.
  \]
\end{definition}

\begin{lemma}
  Let $N \ge 1$. The function $h_N$ satisfies the following properties:
  \begin{enumerate}
    \item $h_N$ is convex on $\mathbb{R}$;
    \item For all $x \in \mathbb{R}$, $0 < h_N(x) \le \frac{e^{-x}}{N}$;
    \item For all $x > 0$, $\frac{e^{-x}}{3N} \le h_N(x)$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  \leanok
  The proof is by calculation of the first and second derivatives and application of the mean value theorem.
\end{proof}
